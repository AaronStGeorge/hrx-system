func.def @branchy(%lhs: i32, %rhs: i32) -> (i32) {
  %c0 = scalar.constant 0 : i32
  %is_zero = scalar.cmpi eq, %lhs, %c0 : i32
  cfg.cond_br %is_zero, ^then, ^else : i1
^then:
  %sum = scalar.addi %rhs, %rhs : i32
  func.return %sum : i32
^else:
  %diff = scalar.subi %lhs, %rhs : i32
  func.return %diff : i32
}
